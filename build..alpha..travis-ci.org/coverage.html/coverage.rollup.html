var is_xlsx = function(d) {
	switch(d[0]) {
		/* CFB container */
		case 0xd0: return true;
		/* XML container (assumed 2003/2004) */
		case 0x3c: return true;
		/* BIFF */
		case 0x09: return true;
		/* Zip container or plaintext */
		case 0x50: return (d[1] == 0x4b && d[2] <= 0x10 && d[3] <= 0x10);

	}
}
